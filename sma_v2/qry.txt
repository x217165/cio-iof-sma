SELECT A.LOB_CODE, A.LOB_DESC, B.SERVICE_CATEGORY_DESC, C.SERVICE_TYPE_ID, C.SERVICE_TYPE_DESC, TO_CHAR(C.SERVICE_TYPE_START_DATE, 'MON-DD-YYYY'), TO_CHAR(C.SERVICE_TYPE_END_DATE, 'MON-DD-YYYY'), D.SERVICE_LEVEL_AGREEMENT_ID, DECODE(UPPER(R.REVENUE_REGION_DESC), 'NATIONAL', 'East', DECODE(R.REVENUE_REGION_DESC, 'US / Int''l', 'East', R.REVENUE_REGION_DESC )) , R.REVENUE_REGION_DESC, D.SERVICE_LEVEL_AGREEMENT_DESC, FROM CRP.LOB A, CRP.SERVICE_LEVEL_AGREEMENT D, CRP.SERVICETYPE_REGION_XREF X, SO.LCODE_REVENUE_REGION R, CRP.SERVICE_CATEGORY B, CRP.SERVICE_TYPE C WHERE A.LOB_ID = B.LOB_ID AND B.SERVICE_CATEGORY_ID = C.SERVICE_CATEGORY_ID AND C.SERVICE_TYPE_ID = X.SERVICE_TYPE_ID(+) AND X.REGION_LCODE = R.REVENUE_REGION_LCODE(+) AND X.SERVICE_LEVEL_AGREEMENT_ID = D.SERVICE_LEVEL_AGREEMENT_ID(+) AND UPPER(C.SERVICE_TYPE_DESC) LIKE '%DRESS%%' AND (TO_DATE('09/30/2004', 'MM/DD/YYYY') BETWEEN C.SERVICE_TYPE_START_DATE AND NVL(C.SERVICE_TYPE_END_DATE, TO_DATE('01/01/3000', 'MM/DD/YYYY'))) AND C.RECORD_STATUS_IND = 'A' ORDER BY A.LOB_DESC ASC, B.SERVICE_CATEGORY_DESC ASC, C.SERVICE_TYPE_DESC ASC, R.REVENUE_REGION_DESC ASC UNION SELECT A.LOB_CODE, A.LOB_DESC, B.SERVICE_CATEGORY_DESC, C.SERVICE_TYPE_ID, C.SERVICE_TYPE_DESC, TO_CHAR(C.SERVICE_TYPE_START_DATE, 'MON-DD-YYYY'), TO_CHAR(C.SERVICE_TYPE_END_DATE, 'MON-DD-YYYY'), D.SERVICE_LEVEL_AGREEMENT_ID, DECODE(UPPER(R.REVENUE_REGION_DESC), 'NATIONAL', 'East', DECODE(R.REVENUE_REGION_DESC, 'US / Int''l', 'East', R.REVENUE_REGION_DESC )) , R.REVENUE_REGION_DESC, D.SERVICE_LEVEL_AGREEMENT_DESC, FROM CRP.LOB A, CRP.SERVICE_LEVEL_AGREEMENT D, CRP.SERVICETYPE_REGION_XREF X, SO.LCODE_REVENUE_REGION R, CRP.V_SERVICE_CATEGORY B, CRP.V_SERVICE_TYPE C WHERE A.LOB_ID = B.LOB_ID AND B.SERVICE_CATEGORY_ID = C.SERVICE_CATEGORY_ID AND C.SERVICE_TYPE_ID = X.SERVICE_TYPE_ID(+) AND X.REGION_LCODE = R.REVENUE_REGION_LCODE(+) AND X.SERVICE_LEVEL_AGREEMENT_ID = D.SERVICE_LEVEL_AGREEMENT_ID(+) AND UPPER(C.SERVICE_TYPE_DESC) LIKE '%DRESS%%' AND (TO_DATE('09/30/2004', 'MM/DD/YYYY') BETWEEN C.SERVICE_TYPE_START_DATE AND NVL(C.SERVICE_TYPE_END_DATE, TO_DATE('01/01/3000', 'MM/DD/YYYY'))) AND C.RECORD_STATUS_IND = 'A' ORDER BY A.LOB_DESC ASC, B.SERVICE_CATEGORY_DESC ASC, C.SERVICE_TYPE_DESC ASC, R.REVENUE_REGION_DESC ASC
